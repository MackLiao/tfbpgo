# Golden URLs for the snapshot-based parity suite (run_parity.sh).
# One URL per line. Lines starting with '#' are comments. Blank lines
# ignored. The '{V}' placeholder is substituted with the artifact version
# read from /api/version at run time, so snapshot filenames are stable
# across artifact bumps (the script hashes the unrendered URL).
#
# Keep this list comprehensive across endpoints — the script's exact-byte
# diff is the regression gate.

/api/version
/api/v/{V}/datasets
/api/v/{V}/regulators?search=YOX&limit=10
/api/v/{V}/regulators/resolve?intersect=callingcards,hackett
/api/v/{V}/regulators/resolve?common=callingcards:hackett
/api/v/{V}/regulators/resolve?regulators=YBR289W,YML007W
/api/v/{V}/binding?regulator=YBR289W&datasets=callingcards
/api/v/{V}/perturbation?regulator=YBR289W&datasets=hackett
/api/v/{V}/comparison/topn?binding=callingcards&perturbation=hackett&top_n=10
/api/v/{V}/comparison/topn?binding=callingcards&perturbation=hackett&top_n=25&effect=0.5
/api/v/{V}/comparison/dto
# Correlation endpoints (A3). The fixture has exactly one binding dataset
# (callingcards) and one perturbation dataset (hackett); dedupeAndCapCSV
# collapses CSV duplicates, so /binding/corr and /perturbation/correlations
# cannot produce a 200 against the fixture (they require >= 2 distinct
# datasets after dedup). The scatter endpoints accept self-pairs without
# dedup, so we cover the A3 surface there.
/api/v/{V}/binding/scatter?regulator=YBR289W&pair=callingcards,callingcards&method=pearson&col=effect
/api/v/{V}/binding/scatter?regulator=YBR289W&pair=callingcards,callingcards&method=spearman&col=effect
/api/v/{V}/perturbation/scatter?regulator=YBR289W&pair=hackett,hackett&method=pearson&col=effect
/api/v/{V}/perturbation/scatter?regulator=YBR289W&pair=hackett,hackett&method=spearman&col=effect
# log10pval measurement (BIND-1/PERT-1). callingcards has no log10p column so
# log10pval resolves to poisson_pval (source="pval"); the Pearson scatter
# applies the server-side -log10(clip(p, 1e-10)) transform (capped at 10),
# while the Spearman scatter returns untransformed ranks. hackett has no
# p-value, so its log10pval resolves to the effect col (source="none", no
# transform) — covering the fallback path.
/api/v/{V}/binding/scatter?regulator=YBR289W&pair=callingcards,callingcards&method=pearson&col=log10pval
/api/v/{V}/binding/scatter?regulator=YBR289W&pair=callingcards,callingcards&method=spearman&col=log10pval
/api/v/{V}/perturbation/scatter?regulator=YBR289W&pair=hackett,hackett&method=pearson&col=log10pval
